This is just one of those things you have to write, apparently. Here is mine, in C++.

#include <iostream>

int main() {
    int _3 {0};
    int _5 {0};

    for (int i = 1; i <= 100; ++i) {
        ++_3;
        ++_5;

        if (_3 == 3) {
            _3 = 0;
            std::cout << "Fizz";
        }

        if (_5 == 5) {
            _5 = 0;
            std::cout << "Buzz";
        }

        if (_3 > 0 && _5 > 0) {
            std::cout << i;
        }

        std::cout << " ";
    }
}
